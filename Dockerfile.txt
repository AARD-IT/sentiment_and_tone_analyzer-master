# Dockerfile content for AI Customer Feedback Analyzer

# 1. Base Image: Use a stable Python image.
FROM python:3.10-slim-buster

# 2. Set the working directory inside the container
WORKDIR /app

# 3. Install System Dependencies: CRITICAL for handling file I/O (Excel/Pandas)
# Install necessary libraries for data processing and locale stability
RUN apt-get update && \
    apt-get install -y --no-install-recommends \
    build-essential \
    libgomp1 \
    && rm -rf /var/lib/apt/lists/*

# 4. Install Python Dependencies
# Copy requirements.txt first to leverage Docker caching.
COPY requirements.txt .
RUN pip install --no-cache-dir -r requirements.txt

# 5. Copy Application Code and Assets
# This copies app.py, assets/, and demo_documents/
COPY . /app

# 6. Expose the port Streamlit uses
EXPOSE 8501

# 7. Command to Run the Application
# Standard Streamlit command.
CMD ["streamlit", "run", "app.py", "--server.port=8501", "--server.address=0.0.0.0"]
